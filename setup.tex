% ============================================================================
% Document Class and Basic Configuration
% ============================================================================
\documentclass[a4paper, 12pt, openany]{book}
% Paper size and font size. Openany ensures that all chapters and similar
% may begin at any page, not only odd pages. For introductory pages and
% appendices we want openany, but for chapter pages in the main content we
% want chapters to begin only on odd pages (right hand side). The book class
% ensures that margins are automatically adjusted such that left hand pages
% are slightly moved to the left and vice versa at the right, which makes
% the thesis very readable and good looking when printed in bound book format.

% ============================================================================
% Character Encoding
% ============================================================================
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}

% ============================================================================
% Chapter and Header Styling
% ============================================================================
\usepackage[Bjarne]{fncychap}
% Fancy chapter style (many more available, like Sonny or Lenny etc.)
\usepackage{fancyhdr}
% Customize the headers

% ============================================================================
% Page Layout and Margins
% ============================================================================
\usepackage[lmargin=1.5in, rmargin=1in, tmargin=1in, bmargin=1in]{geometry}
% Sets the margins for the pages
\setcounter{tocdepth}{2}
% Table of contents number depth for subsections (2 = x.x.x)
\setcounter{secnumdepth}{4}
% Numbering depth for headers for subsections in the text (4 = x.x.x.x)

% ============================================================================
% URLs and Code Listings
% ============================================================================
\usepackage{url}
% Include this if you want to include URLs in the thesis
\usepackage{listings}
% Include this if you want to include code in the thesis

% ============================================================================
% Mathematical Packages
% ============================================================================
\usepackage{amsmath,amssymb}
% Mathematical package
\usepackage{siunitx}
% Includes SI-units

% ============================================================================
% Captions and Floats
% ============================================================================
\usepackage[bf]{caption}
% Makes float captions bold
\usepackage{float}
% Include floats
\usepackage[export]{adjustbox}
% Adjust floats
\usepackage{subfig}
% Include subfigures

% ============================================================================
% Tables
% ============================================================================
\usepackage{array, booktabs}
% Make better tables

% ============================================================================
% Graphics and Figures
% ============================================================================
\usepackage{graphicx}
% Include graphics
\usepackage{chngcntr}
% Change counters for tables, figures etc.
\counterwithin{figure}{section}
% Change counter for figures within sections (also possible per chapter)
\counterwithin{table}{section}
% Change counter for tables within sections

% ============================================================================
% Colors
% ============================================================================
\usepackage{color, xcolor}
% Edit e.g. text colors

% ============================================================================
% Bibliography
% ============================================================================
\usepackage[backend = biber,
            style = numeric,
            date = long,     % Long: 24th Mar. 1997 | Short: 24/03/1997
            sorting = none,
            maxcitenames = 3,   % max names to include before et. al.
            ]{biblatex}
% Customize the look of your citations and bibliography
\addbibresource{bibliography.bib}
% Declare the bibliography resource

% ============================================================================
% Utilities
% ============================================================================
\usepackage{comment}
% Ability to comment out sections in .tex files
\usepackage{afterpage}
% Customize page commands such as below

% ============================================================================
% Custom Commands
% ============================================================================
\newcommand\myemptypage{
    \null
    \thispagestyle{empty}
    \addtocounter{page}{-1}
    \newpage
}
% Sets new page command to insert an empty page without adding to the
% page counter or having a page number
