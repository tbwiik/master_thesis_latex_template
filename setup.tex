\documentclass[b5paper, 11pt, oneside]{report}
% Using report class for simpler formatting suitable for thesis.
% B5 paper (176mm x 250mm), 11pt font, oneside for consistent margins.
\usepackage[english]{babel}

% ============================================================================
% Character Encoding
% ============================================================================
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}

% ============================================================================
% Chapter and Header Styling
% ============================================================================
\usepackage[Bjarne]{fncychap}
% Fancy chapter style (many more available, like Sonny or Lenny etc.)
\usepackage{fancyhdr}
% Customize the headers

% ============================================================================
% Page Layout and Margins
% ============================================================================
\usepackage[b5paper,
            left=0.14\paperwidth,      % ~14% of paper width (≈2.5cm on B5)
            right=0.11\paperwidth,     % ~11% of paper width (≈2cm on B5)

            headheight=1.5\baselineskip, % Relative to line height
            headsep=2ex,               % Relative to x-height

\setcounter{tocdepth}{2}
% Table of contents number depth for subsections (2 = x.x.x)

% Numbering depth for headers for subsections in the text (4 = x.x.x.x)
